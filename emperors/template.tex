\newcommand{\placetextbox}[3]{% 
  \setbox0=\hbox{#3}
  \AddToShipoutPictureFG*{%
    \put(\LenToUnit{#1\paperwidth},\LenToUnit{#2\paperheight}){\vtop{{\null}\makebox[0pt][l]{#3}}}%
  }
}

\newcommand{\doortitle}[1]{\vspace{-0.6cm}\centering \Huge #1}

% \doubleentry{<column number (1 or 2)>}{<row number (1, 2, 3 or 4)>}{<character name>}{<student name>}{<path to image>}
\newcommand{\doubleentry}[5]{%
	\placetextbox{\column{#1}}{\charrow{#2}}{\hspace{0.7cm}\huge #3}
	\placetextbox{\column{#1}}{\studentrow{#2}}{\hspace{0.7cm}\normalsize #4}
	\placetextbox{\column{#1}}{\imagerow{#2}}{\hspace*{-1cm}\includegraphics[width=1.6cm, height=1.6cm]{#5}}
}
