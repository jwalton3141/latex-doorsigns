% \studentleft{name 1}{name 2}{path to image}
% Convenience function to add entry to table
% The image is set on the left of the name
\newcommand{\studentleft}[3]{%
	\addimage{#3} & \thead[cl]{\hspace{-0.1cm}\Huge{#1}\vspace{0.15cm}\\\hspace{-0.2cm}\huge{#2}}
}

% \studentright{name 1}{name 2}{path to image}
% Convenience function to add entry to table
% The image is set on the right of the name
\newcommand{\studentright}[3]{%
	\thead[cr]{\Huge{#1}\vspace{0.15cm}\\\huge{#2}} & \hspace{-0.1cm}\addimage{#3}
}

% \doortitle{room number}
\newcommand{\doortitle}[1]{\centering \Huge #1}

% \background{opacity (0-1)}{path to background image}
\newcommand{\background}[2]{%
        \tikz[remember picture, overlay] \node[opacity=#1, inner sep=0pt] at
	(current page.center){\includegraphics[width=\paperwidth, height=\paperheight]{#2}};
}

% Convenience command \addimage{path to file}
\newcommand{\addimage}[1]{%
	\raisebox{-.45\height}{\includegraphics[width=1.6cm, height=1.6cm]{#1}}%
}

% Stretch space between rows
\renewcommand{\arraystretch}{2.5}
